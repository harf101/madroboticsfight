let audioCtx = null;
let musicOn = true;
let musicPlaying = false;
let bgmGain = null;
let bgmTimeout = null;

let ambientGain = null;

function getAudio() {
  if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
  return audioCtx;
}

function playSound(type) {
  if (!musicOn) return;
  try {
    const ctx = getAudio();
    const now = ctx.currentTime;
    if (type === 'punch') {
      const buf = ctx.createBuffer(1, ctx.sampleRate * 0.12, ctx.sampleRate);
      const d = buf.getChannelData(0);
      for (let i = 0; i < d.length; i++) d[i] = (Math.random() * 2 - 1) * Math.pow(1 - i / d.length, 3);
      const src = ctx.createBufferSource(); src.buffer = buf;
      const g = ctx.createGain(); g.gain.setValueAtTime(0.7, now); g.gain.exponentialRampToValueAtTime(0.001, now + 0.12);
      const f = ctx.createBiquadFilter(); f.type = 'lowpass'; f.frequency.value = 400;
      src.connect(f); f.connect(g); g.connect(ctx.destination); src.start(now);
    } else if (type === 'kick') {
      const o = ctx.createOscillator(); const g = ctx.createGain();
      o.frequency.setValueAtTime(180, now); o.frequency.exponentialRampToValueAtTime(40, now + 0.18);
      g.gain.setValueAtTime(0.9, now); g.gain.exponentialRampToValueAtTime(0.001, now + 0.22);
      o.connect(g); g.connect(ctx.destination); o.start(now); o.stop(now + 0.22);
    } else if (type === 'block') {
      [200, 400, 600].forEach((freq, i) => {
        const o = ctx.createOscillator(); const g = ctx.createGain();
        o.type = 'square'; o.frequency.value = freq;
        g.gain.setValueAtTime(0.15, now + i * 0.01); g.gain.exponentialRampToValueAtTime(0.001, now + 0.15);
        o.connect(g); g.connect(ctx.destination); o.start(now + i * 0.01); o.stop(now + 0.2);
      });
    } else if (type === 'special') {
      const o = ctx.createOscillator(); const g = ctx.createGain();
      const f = ctx.createBiquadFilter(); f.type = 'bandpass'; f.frequency.setValueAtTime(200, now); f.frequency.exponentialRampToValueAtTime(2000, now + 0.5);
      o.type = 'sawtooth'; o.frequency.setValueAtTime(80, now); o.frequency.exponentialRampToValueAtTime(400, now + 0.5);
      g.gain.setValueAtTime(0.8, now); g.gain.exponentialRampToValueAtTime(0.001, now + 0.55);
      o.connect(f); f.connect(g); g.connect(ctx.destination); o.start(now); o.stop(now + 0.6);
    } else if (type === 'mad') {
      [0, 0.07, 0.15].forEach((t, i) => {
        const o = ctx.createOscillator(); const g = ctx.createGain();
        const freqs = [300, 200, 100];
        o.frequency.setValueAtTime(freqs[i], now + t); o.frequency.exponentialRampToValueAtTime(freqs[i] * 0.3, now + t + 0.3);
        g.gain.setValueAtTime(0.7, now + t); g.gain.exponentialRampToValueAtTime(0.001, now + t + 0.35);
        o.connect(g); g.connect(ctx.destination); o.start(now + t); o.stop(now + t + 0.4);
      });
    } else if (type === 'fight') {
      const o = ctx.createOscillator(); const g = ctx.createGain();
      o.type = 'sawtooth'; o.frequency.setValueAtTime(150, now); o.frequency.exponentialRampToValueAtTime(600, now + 0.08); o.frequency.exponentialRampToValueAtTime(200, now + 0.3);
      g.gain.setValueAtTime(0.8, now); g.gain.exponentialRampToValueAtTime(0.001, now + 0.4);
      o.connect(g); g.connect(ctx.destination); o.start(now); o.stop(now + 0.45);
    } else if (type === 'combo') {
      [0, 0.08, 0.16].forEach((t, i) => {
        const o = ctx.createOscillator(); const g = ctx.createGain();
        o.frequency.value = 300 + i * 80;
        g.gain.setValueAtTime(0.5, now + t); g.gain.exponentialRampToValueAtTime(0.001, now + t + 0.1);
        o.connect(g); g.connect(ctx.destination); o.start(now + t); o.stop(now + t + 0.12);
      });
    } else if (type === 'jump') {
      const o = ctx.createOscillator(); const g = ctx.createGain();
      o.type = 'sine'; o.frequency.setValueAtTime(200, now); o.frequency.exponentialRampToValueAtTime(800, now + 0.2);
      g.gain.setValueAtTime(0.3, now); g.gain.exponentialRampToValueAtTime(0.001, now + 0.25);
      o.connect(g); g.connect(ctx.destination); o.start(now); o.stop(now + 0.28);
    } else if (type === 'win') {
      const notes = [523, 659, 784, 1047];
      notes.forEach((freq, i) => {
        const o = ctx.createOscillator(); const g = ctx.createGain();
        o.type = 'square'; o.frequency.value = freq;
        const t = now + i * 0.12;
        g.gain.setValueAtTime(0.3, t); g.gain.exponentialRampToValueAtTime(0.001, t + 0.25);
        o.connect(g); g.connect(ctx.destination); o.start(t); o.stop(t + 0.28);
      });
    } else if (type === 'menu') {
      const o = ctx.createOscillator(); const g = ctx.createGain();
      o.frequency.setValueAtTime(600, now); o.frequency.exponentialRampToValueAtTime(800, now + 0.06);
      g.gain.setValueAtTime(0.2, now); g.gain.exponentialRampToValueAtTime(0.001, now + 0.08);
      o.connect(g); g.connect(ctx.destination); o.start(now); o.stop(now + 0.1);
    }
  } catch(e) {}
}

function playDrum(ctx, dest, time, type) {
  try {
    if (type === 'kick') {
      const o = ctx.createOscillator(); const g = ctx.createGain();
      o.frequency.setValueAtTime(150, time); o.frequency.exponentialRampToValueAtTime(30, time + 0.12);
      g.gain.setValueAtTime(1, time); g.gain.exponentialRampToValueAtTime(0.001, time + 0.15);
      o.connect(g); g.connect(dest); o.start(time); o.stop(time + 0.16);
    } else if (type === 'snare') {
      const buf = ctx.createBuffer(1, ctx.sampleRate * 0.1, ctx.sampleRate);
      const d = buf.getChannelData(0);
      for (let i = 0; i < d.length; i++) d[i] = (Math.random() * 2 - 1) * (1 - i / d.length);
      const src = ctx.createBufferSource(); src.buffer = buf;
      const g = ctx.createGain(); const f = ctx.createBiquadFilter(); f.type = 'highpass'; f.frequency.value = 1000;
      g.gain.setValueAtTime(0.6, time); g.gain.exponentialRampToValueAtTime(0.001, time + 0.1);
      src.connect(f); f.connect(g); g.connect(dest); src.start(time);
    } else if (type === 'hihat') {
      const buf = ctx.createBuffer(1, ctx.sampleRate * 0.04, ctx.sampleRate);
      const d = buf.getChannelData(0);
      for (let i = 0; i < d.length; i++) d[i] = (Math.random() * 2 - 1) * (1 - i / d.length);
      const src = ctx.createBufferSource(); src.buffer = buf;
      const g = ctx.createGain(); const f = ctx.createBiquadFilter(); f.type = 'highpass'; f.frequency.value = 8000;
      g.gain.setValueAtTime(0.2, time); g.gain.exponentialRampToValueAtTime(0.001, time + 0.04);
      src.connect(f); f.connect(g); g.connect(dest); src.start(time);
    }
  } catch(e) {}
}

function startBattleMusic() {
  if (!musicOn || musicPlaying) return;
  try {
    const ctx = getAudio();
    musicPlaying = true;
    bgmGain = ctx.createGain(); bgmGain.gain.value = 0.12; bgmGain.connect(ctx.destination);
    scheduleBGM(ctx.currentTime);
  } catch(e) {}
}

function stopBattleMusic() {
  musicPlaying = false;
  if (bgmTimeout) clearTimeout(bgmTimeout);
  if (bgmGain) {
    try { if (audioCtx) bgmGain.gain.setTargetAtTime(0, audioCtx.currentTime, 0.3); } catch(e) {}
    setTimeout(() => { try { bgmGain.disconnect(); } catch(e) {} bgmGain = null; }, 500);
  }
}

function scheduleBGM(startTime) {
  if (!musicPlaying || !bgmGain) return;
  try {
    const ctx = getAudio(); const bpm = 140; const beat = 60 / bpm; const bar = beat * 4; const loopLen = bar * 4;
    [0, beat * 2].forEach(t => playDrum(ctx, bgmGain, startTime + t, 'kick'));
    [beat, beat * 3].forEach(t => playDrum(ctx, bgmGain, startTime + t, 'snare'));
    for (let i = 0; i < 8; i++) playDrum(ctx, bgmGain, startTime + i * beat * 0.5, 'hihat');
    playDrum(ctx, bgmGain, startTime + beat * 0.5, 'kick');
    [bar, bar + beat * 2, bar + beat * 0.75].forEach(t => playDrum(ctx, bgmGain, startTime + t, 'kick'));
    [bar + beat, bar + beat * 3].forEach(t => playDrum(ctx, bgmGain, startTime + t, 'snare'));
    for (let i = 0; i < 8; i++) playDrum(ctx, bgmGain, startTime + bar + i * beat * 0.5, 'hihat');
    const bassNotes = [55, 55, 73, 73, 65, 65, 49, 49];
    bassNotes.forEach((freq, i) => {
      const o = ctx.createOscillator(); const g = ctx.createGain();
      const f = ctx.createBiquadFilter(); f.type = 'lowpass'; f.frequency.value = 300;
      o.type = 'sawtooth'; o.frequency.value = freq;
      const t = startTime + i * beat * 0.5;
      g.gain.setValueAtTime(0, t); g.gain.linearRampToValueAtTime(0.5, t + 0.02);
      g.gain.setValueAtTime(0.5, t + beat * 0.4); g.gain.linearRampToValueAtTime(0, t + beat * 0.5);
      o.connect(f); f.connect(g); g.connect(bgmGain); o.start(t); o.stop(t + beat * 0.52);
    });
    const melNotes = [220, 246, 261, 293, 329, 293, 261, 220, 246, 293, 329, 440, 392, 329, 293, 261];
    melNotes.forEach((freq, i) => {
      const o = ctx.createOscillator(); const g = ctx.createGain();
      o.type = 'square'; o.frequency.value = freq;
      const t = startTime + bar * 2 + i * beat * 0.5;
      g.gain.setValueAtTime(0, t); g.gain.linearRampToValueAtTime(0.12, t + 0.02);
      g.gain.setValueAtTime(0.12, t + beat * 0.35); g.gain.linearRampToValueAtTime(0, t + beat * 0.5);
      o.connect(g); g.connect(bgmGain); o.start(t); o.stop(t + beat * 0.52);
    });
    const remaining = (startTime + loopLen - ctx.currentTime) * 1000 - 200;
    bgmTimeout = setTimeout(() => { if (musicPlaying && bgmGain) scheduleBGM(startTime + loopLen); }, Math.max(100, remaining));
  } catch(e) { musicPlaying = false; }
}

function startAmbientMusic() {
  if (!musicOn || ambientPlaying) return;
  try {
    const ctx = getAudio();
    ambientPlaying = true;
    ambientGain = ctx.createGain(); ambientGain.gain.value = 0.08; ambientGain.connect(ctx.destination);
    scheduleAmbient(ctx.currentTime);
  } catch(e) {}
}

function stopAmbientMusic() {
  ambientPlaying = false;
  if (ambientTimeout) clearTimeout(ambientTimeout);
  if (ambientGain) {
    try { if (audioCtx) ambientGain.gain.setTargetAtTime(0, audioCtx.currentTime, 0.5); } catch(e) {}
    setTimeout(() => { try { ambientGain.disconnect(); } catch(e) {} ambientGain = null; }, 600);
  }
}

function scheduleAmbient(startTime) {
  if (!ambientPlaying || !ambientGain) return;
  try {
    const ctx = getAudio(); const beat = 60 / 90; const loopLen = beat * 8;
    const chords = [[110, 138, 165], [98, 123, 147], [117, 147, 175], [87, 110, 131]];
    chords.forEach((chord, ci) => {
      chord.forEach(freq => {
        const o = ctx.createOscillator(); const g = ctx.createGain();
        o.type = 'sine'; o.frequency.value = freq;
        const t = startTime + ci * beat * 2;
        g.gain.setValueAtTime(0, t); g.gain.linearRampToValueAtTime(0.3, t + beat * 0.5);
        g.gain.setValueAtTime(0.3, t + beat * 1.5); g.gain.linearRampToValueAtTime(0, t + beat * 2);
        o.connect(g); g.connect(ambientGain); o.start(t); o.stop(t + beat * 2.1);
      });
    });
    for (let i = 0; i < 2; i++) playDrum(ctx, ambientGain, startTime + i * beat * 4, 'kick');
    const remaining = (startTime + loopLen - ctx.currentTime) * 1000 - 200;
    ambientTimeout = setTimeout(() => { if (ambientPlaying && ambientGain) scheduleAmbient(startTime + loopLen); }, Math.max(100, remaining));
  } catch(e) { ambientPlaying = false; }
}

function toggleMusic() {

  const icon = document.getElementById('music-icon');

  const gameBtn = document.getElementById('game-music-btn');

  if (label) label.textContent = musicOn ? 'MUSIC: ON' : 'MUSIC: OFF';
  if (gameBtn) gameBtn.textContent = musicOn ? '🔊' : '🔇';
  if (!musicOn) { stopBattleMusic(); stopAmbientMusic(); }
  else {
    const gameVisible = document.getElementById('game-screen').classList.contains('active');
    const homeVisible = document.getElementById('home-screen').classList.contains('active');
    if (gameVisible) startBattleMusic();
    else if (homeVisible) startAmbientMusic();
  }
}

function showScreen(id) {
  document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
  document.getElementById(id).classList.add('active');
  // Only touch audio if AudioContext is already running (requires prior user gesture)
  if (!audioCtx) return;
  try {
    if (id === 'home-screen') { stopBattleMusic(); setTimeout(startAmbientMusic, 200); }
    else if (id === 'game-screen') { stopAmbientMusic(); }
    else { stopAmbientMusic(); stopBattleMusic(); }
  } catch(e) {}
}

function goToCharSelect(mode) {
  if (audioCtx) { try { playSound('menu'); stopBattleMusic(); stopAmbientMusic(); } catch(e) {} }
  gameMode = mode;

  document.getElementById('p2-sel-name').textContent = '???';

function renderCharGrid() {
  const grid = document.getElementById('fighter-grid');
  grid.innerHTML = '';
  CHARACTERS.forEach((char, i) => {
    const div = document.createElement('div');
    div.className = 'fighter-card';
    div.id = 'card-' + i;
    if (p1Char && p1Char.id === i) div.classList.add('selected');
    if (p2Char && p2Char.id === i) div.classList.add('p2-selected');
    div.innerHTML = '<div class="fighter-avatar">' + drawFighter(char) + '</div><div class="fighter-name" style="color:' + char.color + '">' + char.name + '</div><div class="fighter-power-badge">⚡ ' + char.special + '</div>';
    div.addEventListener('click', () => selectChar(i));
    div.addEventListener('touchstart', (e) => { e.preventDefault(); selectChar(i); }, { passive: false });
    grid.appendChild(div);
  });
}

function selectChar(i) {

}

function startTimer() {

    if (gameState.timeLeft <= 10) document.getElementById('timer-display').style.color = '#e74c3c';
    else document.getElementById('timer-display').style.color = '#f39c12';
    if (gameState.timeLeft <= 0) timeUp();
  }, 1000);
}

let lastAction = 0;
const ACTION_COOLDOWN = 300;

function doAction(type) {
  if (!gameState.gameRunning) return;
  const now = Date.now();
  if (now - lastAction < ACTION_COOLDOWN) return;
  lastAction = now;

  if (type === 'block') {
    playSound('block');
    gameState.p1Blocking = true;
    animateFighter('p1', 'block');
    setTimeout(() => { gameState.p1Blocking = false; }, 600);
    return;
  }

  let damage = 0, hitText = '', powerGain = 0;
  if (type === 'punch') {
    playSound('punch');
    damage = 8 + Math.floor(Math.random() * 7);
    hitText = getHitText('punch'); powerGain = 12;
    animateFighter('p1', 'attack');
  } else if (type === 'kick') {
    playSound('kick');
    damage = 11 + Math.floor(Math.random() * 9);
    hitText = getHitText('kick'); powerGain = 15;
    animateFighter('p1', 'attack');
  } else if (type === 'combo') {
    playSound('combo');
    damage = 15 + Math.floor(Math.random() * 10);
    hitText = 'COMBO!'; powerGain = 20;
    animateFighter('p1', 'attack');
  }

  if (gameState.p2Blocking && Math.random() > 0.4) {
    damage = Math.floor(damage * 0.15);
    hitText = 'BLOCKED';
    showHitText(hitText, 'p2', '#00bcd4');
  } else {
    applyDamage('p2', damage);
    showHitText(hitText, 'p2', p1Char.glowColor);
    spawnBlood('p2');
    flashScreen();
  }

  gainPower('p1', powerGain);
  setTimeout(() => aiCounter(), 200 + Math.random() * 500);
}

function useSpecial() {
  if (!gameState.gameRunning) return;
  if (gameState.p1Power < 100) {
    showHitText('NEED MORE POWER!', 'p1', '#888');
    return;
  }
  playSound('special');
  gameState.p1Power = 0;
  updatePowerBars();
  document.getElementById('special-ready-text').textContent = 'FILL POWER BAR';
  document.getElementById('special-ready-text').style.color = '#555';
  const damage = 35 + Math.floor(Math.random() * 20);
  animateFighter('p1', 'special');
  showAnnouncer(p1Char.special + '!', 'special', 1200);
  setTimeout(() => {
    applyDamage('p2', damage);
    spawnBlood('p2');
    flashScreen();
    showHitText('💥 ' + damage, 'p2', p1Char.glowColor);
  }, 400);
}

function pressBtn(dir, down) {
  pressedKeys[dir] = down;
  if (!gameState.gameRunning) return;
  const btn = document.getElementById('btn-' + dir);
  if (btn) down ? btn.classList.add('pressed') : btn.classList.remove('pressed');
  if (down) {
    if (dir === 'up') { playSound('jump'); animateFighter('p1', 'jump'); gainPower('p1', 5); }
    else if (dir === 'left' || dir === 'right') { gainPower('p1', 2); }
    else if (dir === 'down') { gameState.p1Blocking = true; animateFighter('p1', 'block'); setTimeout(() => { gameState.p1Blocking = false; }, 500); }
  }
}

function gainPower(player, amount) {
  if (player === 'p1') {
    gameState.p1Power = Math.min(100, gameState.p1Power + amount);
    if (gameState.p1Power >= 100) {
      document.getElementById('special-ready-text').textContent = '⚡ SPECIAL READY!';
      document.getElementById('special-ready-text').style.color = '#a855f7';
    }
  } else {
    gameState.p2Power = Math.min(100, gameState.p2Power + amount);
  }
  updatePowerBars();
}

function startAI() {
  gameState.aiInterval = setInterval(() => {
    if (!gameState.gameRunning) return;
    const r = Math.random();
    if (r < 0.15) aiCounter();
  }, 1200);
}

function aiCounter() {
  if (!gameState.gameRunning) return;
  const actions = ['punch', 'punch', 'kick', 'combo', 'block'];
  const action = actions[Math.floor(Math.random() * actions.length)];
  if (action === 'block') {
    gameState.p2Blocking = true; animateFighter('p2', 'block');
    setTimeout(() => { gameState.p2Blocking = false; }, 500);
    return;
  }
  const damage = action === 'punch' ? 6 + Math.floor(Math.random() * 6) : action === 'kick' ? 9 + Math.floor(Math.random() * 8) : 12 + Math.floor(Math.random() * 8);
  animateFighter('p2', 'attack'); gainPower('p2', 12);
  if (gameState.p1Blocking && Math.random() > 0.5) {
    showHitText('BLOCKED', 'p1', '#00bcd4');
  } else {
    applyDamage('p1', damage);
    showHitText(getHitText(action), 'p1', p2Char.glowColor);
    spawnBlood('p1');
  }
  if (gameState.p2Power >= 100 && Math.random() > 0.5) {
    gameState.p2Power = 0; updatePowerBars();
    const sDmg = 30 + Math.floor(Math.random() * 20);
    setTimeout(() => {
      animateFighter('p2', 'special');
      applyDamage('p1', sDmg);
      showHitText('⚡ ' + sDmg, 'p1', p2Char.glowColor);
      flashScreen();
    }, 200);
  }
}

function getHitText(type) {
  const punches = ['MAD PUNCH!', 'MAD HIT!', 'MAD POW!', 'MAD BAM!', 'MAD CRACK!'];
  const kicks = ['KICK!', 'SLAM!', 'BASH!', 'CRUSH!', 'WHAM!'];
  if (type === 'punch') return punches[Math.floor(Math.random() * punches.length)];
  if (type === 'kick') return kicks[Math.floor(Math.random() * kicks.length)];
  return 'COMBO!';
}

// ============================================================
function applyDamage(player, damage) {
  if (player === 'p1') { gameState.p1HP = Math.max(0, gameState.p1HP - damage); animateFighter('p1', 'hurt'); gainPower('p1', 8); }
  else { gameState.p2HP = Math.max(0, gameState.p2HP - damage); animateFighter('p2', 'hurt'); gainPower('p2', 8); }
  updateHealthBars();
  if (gameState.p1HP <= 0 || gameState.p2HP <= 0) roundEnd();
}

function updateHealthBars() {
  const p1Bar = document.getElementById('p1-health-bar');
  const p2Bar = document.getElementById('p2-health-bar');
  p1Bar.style.width = gameState.p1HP + '%';
  p2Bar.style.width = gameState.p2HP + '%';
  const setColor = (bar, hp) => { bar.classList.remove('low','medium'); if (hp <= 25) bar.classList.add('low'); else if (hp <= 50) bar.classList.add('medium'); };
  setColor(p1Bar, gameState.p1HP); setColor(p2Bar, gameState.p2HP);
}

function updatePowerBars() {
  document.getElementById('p1-power-bar').style.width = gameState.p1Power + '%';
  document.getElementById('p2-power-bar').style.width = gameState.p2Power + '%';
}

// ============================================================
// ROUND / MATCH END
// ============================================================
function roundEnd() {
  gameState.gameRunning = false;
  clearInterval(gameState.timerInterval);
  clearInterval(gameState.aiInterval);
  let winner;
  if (gameState.p1HP <= 0 && gameState.p2HP <= 0) winner = 'DRAW';
  else if (gameState.p1HP <= 0) { winner = 'p2'; gameState.p2Wins++; }
  else { winner = 'p1'; gameState.p1Wins++; }

  if (winner === 'DRAW') showAnnouncer('DRAW!', 'ko', 2000);
  else { showAnnouncer('$MAD!', 'ko', 2000); playSound('mad'); }

  setTimeout(() => {
    if (gameState.p1Wins >= 2 || gameState.p2Wins >= 2) {
      stopBattleMusic();
      showFinalWin(winner);
      playSound('win');
    } else {
      gameState.round++;
      document.getElementById('round-display').textContent = 'ROUND ' + gameState.round;
      resetRound();
      document.getElementById('p1-sprite').innerHTML = drawFighter(p1Char);
      document.getElementById('p2-sprite').innerHTML = drawFighter(p2Char);
      showAnnouncer('ROUND ' + gameState.round, 'fight', 1500);
      setTimeout(() => { gameState.gameRunning = true; startTimer(); if (gameMode === 'ai') startAI(); }, 1500);
    }
  }, 2200);
}

function timeUp() {
  if (gameState.p1HP > gameState.p2HP) { gameState.p1Wins++; showAnnouncer('TIME!', 'ko', 2000); setTimeout(() => showFinalWin('p1'), 2200); }
  else if (gameState.p2HP > gameState.p1HP) { gameState.p2Wins++; showAnnouncer('TIME!', 'ko', 2000); setTimeout(() => showFinalWin('p2'), 2200); }
  else { showAnnouncer('DRAW!', 'ko', 2000); setTimeout(() => goToCharSelect('local'), 2500); }
}

function showFinalWin(player) {
  clearInterval(gameState.timerInterval); clearInterval(gameState.aiInterval);
  gameState.gameRunning = false;
  const winnerChar = player === 'p1' ? p1Char : p2Char;
  document.getElementById('win-title').textContent = player === 'DRAW' ? 'DRAW!' : 'WINNER!';
  document.getElementById('winner-name').textContent = player === 'DRAW' ? 'BOTH FIGHTERS' : winnerChar.name;
  document.getElementById('winner-name').style.color = player === 'DRAW' ? '#aaa' : winnerChar.color;
  document.getElementById('round-wins-text').textContent = 'P1: ' + gameState.p1Wins + ' — P2: ' + gameState.p2Wins;
  showScreen('win-screen');
}

function doSignup() {
  const u = document.getElementById('signup-username').value.trim();
  const p = document.getElementById('signup-password').value;
  const p2 = document.getElementById('signup-password2').value;
  if (!u || u.length < 3) return setStatus('signup-status', '⚠ Username must be at least 3 characters!', 'error');
  if (!/^[a-zA-Z0-9_]+$/.test(u)) return setStatus('signup-status', '⚠ Only letters, numbers and _ allowed!', 'error');
  if (!p || p.length < 4) return setStatus('signup-status', '⚠ Password must be at least 4 characters!', 'error');
  if (p !== p2) return setStatus('signup-status', '⚠ Passwords do not match!', 'error');
  const users = getUsers();
  if (users[u.toLowerCase()]) return setStatus('signup-status', '⚠ Username already taken!', 'error');
  users[u.toLowerCase()] = { username: u, password: btoa(p), wins: 0, losses: 0, joined: Date.now() };
  saveUsers(users);
  setStatus('signup-status', '✅ Account created! Logging you in...', 'connected');
  setTimeout(() => loginAs(u), 1000);
}

function doLogin() {
  const u = document.getElementById('login-username').value.trim();
  const p = document.getElementById('login-password').value;
  if (!u || !p) return setStatus('login-status', '⚠ Enter username and password!', 'error');
  const users = getUsers();
  const rec = users[u.toLowerCase()];
  if (!rec) return setStatus('login-status', '⚠ Username not found!', 'error');
  if (rec.password !== btoa(p)) return setStatus('login-status', '⚠ Wrong password!', 'error');
  setStatus('login-status', '✅ Welcome back, ' + rec.username + '!', 'connected');
  setTimeout(() => loginAs(rec.username), 800);
}

function loginAs(username) {
  currentUser = username;
  const online = getOnline();
  online[username.toLowerCase()] = { username, lastSeen: Date.now() };
  saveOnline(online);
  document.getElementById('lobby-username-display').textContent = username.toUpperCase();
  const users = getUsers(); const rec = users[username.toLowerCase()];
  if (rec) document.getElementById('lobby-record').textContent = 'W: ' + rec.wins + '  L: ' + rec.losses;
  showScreen('lobby-screen');
  refreshLobby();
  if (lobbyInterval) clearInterval(lobbyInterval);
  lobbyInterval = setInterval(refreshLobby, 5000);
}

function logoutUser() {
  if (currentUser) { const online = getOnline(); delete online[currentUser.toLowerCase()]; saveOnline(online); }
  currentUser = null;
  if (lobbyInterval) clearInterval(lobbyInterval);
  showScreen('home-screen');
}

function refreshLobby() {
  if (!currentUser) return;
  const online = getOnline();
  online[currentUser.toLowerCase()] = { username: currentUser, lastSeen: Date.now() };
  const now = Date.now();
  Object.keys(online).forEach(k => { if (now - online[k].lastSeen > 30000) delete online[k]; });
  saveOnline(online);
  const list = document.getElementById('online-players-list');
  const others = Object.values(online).filter(p => p.username.toLowerCase() !== currentUser.toLowerCase());
  if (others.length === 0) {
    list.innerHTML = '<div style="color:#555;padding:10px;">NO OTHER PLAYERS ONLINE<br><span style="font-size:10px;color:#333;">SHARE THE GAME LINK TO INVITE FRIENDS!</span></div>';
  } else {
    list.innerHTML = others.map(p => '<div style="display:flex;justify-content:space-between;align-items:center;padding:8px 0;border-bottom:1px solid #1a1a1a;"><span style="color:#2ecc71;font-weight:700;letter-spacing:1px;">🟢 ' + p.username.toUpperCase() + '</span><button onclick="quickChallenge(\'' + p.username + '\')" style="background:rgba(192,57,43,0.3);border:1px solid #c0392b;color:#e74c3c;font-size:10px;padding:4px 10px;border-radius:4px;cursor:pointer;letter-spacing:1px;">CHALLENGE</button></div>').join('');
  }
  const allChallenges = getChallenges().filter(c => c.to.toLowerCase() === currentUser.toLowerCase() && now - c.time < 60000);
  const incomingEl = document.getElementById('incoming-challenges');
  if (allChallenges.length === 0) {
    incomingEl.innerHTML = '<div style="color:#555;padding:8px;">NO CHALLENGES YET</div>';
  } else {
    incomingEl.innerHTML = allChallenges.map(c => '<div style="display:flex;justify-content:space-between;align-items:center;padding:8px 0;border-bottom:1px solid #1a1a1a;"><span style="color:var(--gold);font-size:12px;">⚔️ ' + c.from.toUpperCase() + '</span><div style="display:flex;gap:5px;"><button onclick="acceptChallenge(\'' + c.from + '\',\'' + c.id + '\')" style="background:rgba(39,174,96,0.3);border:1px solid #27ae60;color:#27ae60;font-size:10px;padding:4px 8px;border-radius:4px;cursor:pointer;">ACCEPT</button><button onclick="declineChallenge(\'' + c.id + '\')" style="background:rgba(192,57,43,0.2);border:1px solid #444;color:#888;font-size:10px;padding:4px 8px;border-radius:4px;cursor:pointer;">DECLINE</button></div></div>').join('');
  }
}

function quickChallenge(username) { document.getElementById('challenge-target').value = username; sendChallenge(); }

function sendChallenge() {
  const target = document.getElementById('challenge-target').value.trim();
  if (!target) return setStatus('challenge-status', '⚠ Enter a username!', 'error');
  if (target.toLowerCase() === currentUser.toLowerCase()) return setStatus('challenge-status', '⚠ You cannot challenge yourself!', 'error');
  const online = getOnline();
  if (!online[target.toLowerCase()]) return setStatus('challenge-status', '⚠ ' + target + ' is not online!', 'error');
  const challenges = getChallenges();
  const id = 'ch_' + Date.now() + '_' + Math.random().toString(36).slice(2,6);
  challenges.push({ id, from: currentUser, to: target, time: Date.now() });
  saveChallenges(challenges);
  setStatus('challenge-status', '✅ Challenge sent to ' + target.toUpperCase() + '! Waiting...', 'waiting');
}

function acceptChallenge(fromUser, challengeId) {
  const remaining = getChallenges().filter(c => c.id !== challengeId);
  saveChallenges(remaining);
  gameMode = 'ai';
  setStatus('challenge-status', '✅ Challenge accepted! Loading arena...', 'connected');
  setTimeout(() => { goToCharSelect('ai'); }, 1000);
}

function declineChallenge(challengeId) {
  const remaining = getChallenges().filter(c => c.id !== challengeId);
  saveChallenges(remaining); refreshLobby();
}
